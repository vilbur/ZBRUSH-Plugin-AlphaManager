/** Resize Alpha buttons to fit into UI
  *
  */
[RoutineDef, resizeAlphaButton
,
	[If, [StrLength, $alpha_id] == 1
	, // THEN get two digit
		[VarSet, $alpha_name, [StrMerge, "Alpha 0", $alpha_id] ]
	, // ELSE get single digit
		[VarSet, $alpha_name, [StrMerge, "Alpha ", $alpha_id] ]
	]

	[IModify, [StrMerge, "Alpha: ", $alpha_name], 86, 0]

, // PARAMS
	$alpha_id
]
